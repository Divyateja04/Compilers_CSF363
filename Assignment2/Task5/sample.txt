program ArrayAverage;
var
  numbers: array[1..5] of Integer;
  i, a, b: Integer;
  average : real;
  c: char;
begin
  i := 100;
  average := 0.5;
  c := 'c';

  write("Enter 5 integer values:", i, i);

  if i > 0 then
  begin
    a := 0;
  end;

  for i := 1 to 5 do
  begin
    a := a + i;
    average := average / 2;
  end;

  while a > 0 do
  begin
    a := a - 1;
    average := average * 2;
  end;

  a := -1;
  if a < 0 then
  begin
    a := 0;
  end;
end.
